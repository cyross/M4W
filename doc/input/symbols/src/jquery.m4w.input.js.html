1 /**
  2  * @fileOverview Miyako4Web(Miyako for Web a.k.a M4W/m4w) Input Extension<br>
  3  * M4W用デバイス入力対応拡張プラグイン
  4  * このファイルを読み込むと、window.m4wオブジェクトに以下のプロパティが追加される<br>
  5  * window.m4w.Input<br>
  6  *
  7  * @name Miyako for Web(M4W) Input extention
  8  * @author Cyross Makoto (サイロス誠)
  9  * @version 0.0.1
 10  * @revision 1

 12  * @license MIT License (MITライセンス)
 13  */

 15 (function($){
 16   /**
 17    * @namespace 各種クラスの外部アクセス用名前空間<br>
 18    * 以下の関数が利用可能
 19    * <ul>
 20    * <li>can_touch</li>
 21    * <li>is_windows8</li>
 22    * <li>get_xy_mouse</li>
 23    * <li>get_xy_touch</li>
 24    * <li>get_xy_multitouch</li>
 25    * <li>get_xy</li>
 26    * </ul>
 27    * また、以下のプロパティが利用可能
 28    * <ul>
 29    * <li>vars(入力デバイス関連情報共有オブジェクト<br>初期は空のオブジェクト)</li>
 30    * </ul>
 31    * @example if((window).m4w.Input.can_touch()){ ... }
 32   */
 33   Input = {
 34     /**
 35      * @property 入力デバイス関連情報共有オブジェクト<br>初期は空のオブジェクト
 36      */
 37     vars: {},
 38     start_events: {touch: "touchstart", mouse: "mousedown"},
 39     move_events: {touch: "touchmove", mouse: "mousemove"},
 40     end_events: {touch: "touchend", mouse: "mouseup"},
 41     click_events: {touch: "touchstart", mouse: "click"},
 42     event_mode: "mouse"
 43   };
 44 
 45   /**
 46    * User Agent から、タッチデバイスかどうか判別する<br>
 47    * 使用している端末・OSが以下の時、trueとなる
 48    * <ul>
 49    * <li>イベントにontouchstartが用意されている
 50    * <li>iPhone
 51    * <li>iPad
 52    * <li>iPod Touch
 53    * <li>Android端末
 54    * <li>Windows Phone
 55    * <li>Windows 8(ディスプレイがタッチ対応かどうかにかかわらず)
 56    * </ul>
 57    * @return タッチデバイスの時はtrue、それ以外はfalse
 58    */
 59   Input.can_touch = function(){
 60     var user_agent = window.navigator.userAgent.toLowerCase();
 61     var agents = ["iphone", "ipad", "ipod", "android", "windowsphone", "windows nt 6.2"];
 62 
 63     if(document.ontouchstart !== undefined){ return true; }
 64 
 65     for(var i=0; i<agents.length; i++){
 66       if(user_agent.indexOf(agents[i]) > -1){ return true; }
 67     }
 68     // その他
 69     return false;
 70   };
 71 
 72   /**
 73    * User Agent から、使用しているOSがWindows 8かどうか判別する<br>
 74    * @return Windows 8の時はtrue、それ以外はfalse
 75    */
 76   Input.is_windows8 = function(){
 77     var user_agent = window.navigator.userAgent.toLowerCase();
 78     if(user_agent.indexOf("windows nt 6.2") > -1){ return true; }
 79     // その他
 80     return false;
 81   };
 82 
 83   /**
 84    * 現在のマウスポインタの位置を取得する<br>
 85    * mousestart, mousemoveなどのイベントハンドリング時に使用する<br>
 86    * $areaで示したブロック(jQueryオブジェクト)の左上を[0,0]とする
 87    * @param ev 取得に使用するイベントオブジェクト
 88    * @param ev 取得に使用するイベントオブジェクト
 89    * @return [x,y]で示す配列
 90    */
 91   Input.get_xy_mouse = function($area, ev){
 92     var offset = $area.offset();
 93     return [ev.pageX - offset.left, ev.pageY - offset.top];
 94   };
 95 
 96   /**
 97    * 現在タッチしている位置を取得する<br>
 98    * touchstart, touchmoveなどのイベントハンドリング時に使用する<br>
 99    * $areaで示したブロック(jQueryオブジェクト)の左上を[0,0]とする
100    * @param ev 取得に使用するイベントオブジェクト
101    * @param ex 取得に使用するイベントオブジェクト
102    * @return [x,y]で示す配列
103    */
104   Input.get_xy_touch = function($area, ev){
105     var offset = $area.offset();
106     var touch = ev.originalEvent.targetTouches[0];
107     return [touch.pageX - offset.left, touch.pageY - offset.top];
108   };
109 
110   /**
111    * 現在全ての指がタッチしている位置を取得する<br>
112    * touchstart, touchmoveなどのイベントハンドリング時に使用する<br>
113    * $areaで示したブロック(jQueryオブジェクト)の左上を[0,0]とする
114    * @param ev 取得に使用するイベントオブジェクト
115    * @param ex 取得に使用するイベントオブジェクト
116    * @return [x,y]で示す配列の配列
117    */
118   Input.get_xy_multitouch = function($area, ev){
119     var result = [];
120     var offset = $area.offset();
121     var touches = ev.originalEvent.targetTouches;
122     for(var i=0; i<touches.length; i++){
123         var touch = touches[i];
124         result.push([touch.pageX - offset.left, touch.pageY - offset.top]);
125     }
126     return result;
127   };
128 
129   /**
130    * イベントモードを切り替える<br>
131    * タッチ対応かマウス対応可を選択<br>
132    * 初期値は”mouse”<br>
133    * @param mode イベントモード<br>"touch"か"mouse”のみ指定可能<br>その他はfalseが返る
134    * @return mode引数が正しければtrue、正しくなければfalse
135    */
136   Input.set_event_mode = function(mode){
137     if(!mode in Input.start_events){ return false; }
138     Input.event_mode = mode;
139     if(mode == "touch"){
140       Input.get_xy = Input.get_xy_touch;
141     }
142     else{
143       Input.get_xy = Input.get_xy_mouse;
144     }
145     return true;
146   }
147 
148   /**
149    * クリックイベント名を取得<br>
150    * @return mouseのときは"click"、touchのときは"touchstart"
151    */
152   Input.click_event_name = function(){
153     return Input.click_events[Input.event_mode];
154   }
155 
156   /**
157    * 開始イベント名を取得<br>
158    * @return mouseのときは"mousestart"、touchのときは"touchstart"
159    */
160   Input.start_event_name = function(){
161     return Input.start_events[Input.event_mode];
162   }
163 
164   /**
165    * 移動中イベント名を取得<br>
166    * @return mouseのときは"mousemove"、touchのときは"touchmove"
167    */
168   Input.move_event_name = function(){
169     return Input.move_events[Input.event_mode];
170   }
171 
172   /**
173    * 終了イベント名を取得<br>
174    * @return mouseのときは"mouseend"、touchのときは"touchend"
175    */
176   Input.end_event_name = function(){
177     return Input.end_events[Input.event_mode];
178   }
179 
180   /**
181    * @property event_modeによって、get_xy_touch、get_xy_mouseを切り替える
182    */
183   Input.get_xy = Input.get_xy_mouse;
184 
185   window.m4w = $.extend({Input: Input}, window.m4w);
186 })(jQuery);
