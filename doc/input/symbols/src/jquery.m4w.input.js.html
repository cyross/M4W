  2  * @fileOverview Miyako4Web(Miyako for Web a.k.a M4W/m4w) Input Extension<br>
  3  * M4W用デバイス入力対応拡張プラグイン
  4  * このファイルを読み込むと、window.m4wオブジェクトに以下のプロパティが追加される<br>
  5  * window.m4w.Input<br>
  6  *
  7  * @name Miyako for Web(M4W) Input extention
  8  * @author Cyross Makoto (サイロス誠)

 15 (function($){
 16   /**
 17    * @namespace 各種クラスの外部アクセス用名前空間<br>
 18    * 以下の関数が利用可能
 19    * <ul>
 20    * <li>can_touch</li>
 21    * <li>is_windows8</li>
 22    * <li>get_xy_mouse</li>
 23    * <li>get_xy_touch</li>
 24    * <li>get_xy_multitouch</li>
 25    * <li>get_xy</li>
 26    * </ul>
 27    * また、以下のプロパティが利用可能
 28    * <ul>
 29    * <li>vars(入力デバイス関連情報共有オブジェクト<br>初期は空のオブジェクト)</li>
 30    * </ul>
 31    * @example if((window).m4w.Input.can_touch()){ ... }
 32   */
 33   Input = {
 34     /**
 35      * @property 入力デバイス関連情報共有オブジェクト<br>初期は空のオブジェクト
 36      */
 37     vars: {}
 38   };
 39 
 40   /**
 41    * User Agent から、タッチデバイスかどうか判別する<br>
 42    * 使用している端末・OSが以下の時、trueとなる
 43    * <ul>
 44    * <li>イベントにontouchstartが用意されている
 45    * <li>iPhone
 46    * <li>iPad
 47    * <li>iPod Touch
 48    * <li>Android端末
 49    * <li>Windows Phone
 50    * <li>Windows 8(ディスプレイがタッチ対応かどうかにかかわらず)
 51    * </ul>
 52    * @return タッチデバイスの時はtrue、それ以外はfalse
 53    */
 54   Input.can_touch = function(){
 55     var user_agent = window.navigator.userAgent.toLowerCase();
 56     var agents = ["iphone", "ipad", "ipod", "android", "windowsphone", "windows nt 6.2"];
 57 
 58     if(document.ontouchstart !== undefined){ return true; }
 59 
 60     for(var i=0; i<agents.length; i++){
 61       if(user_agent.indexOf(agents[i]) > -1){ return true; }
 62     }
 63     // その他
 64     return false;
 65   };
 66 
 67   /**
 68    * User Agent から、使用しているOSがWindows 8かどうか判別する<br>
 69    * @return Windows 8の時はtrue、それ以外はfalse
 70    */
 71   Input.is_windows8 = function(){
 72     var user_agent = window.navigator.userAgent.toLowerCase();
 73     if(user_agent.indexOf("windows nt 6.2") > -1){ return true; }
 74     // その他
 75     return false;
 76   };
 77 
 78   /**
 79    * 現在のマウスポインタの位置を取得する<br>
 80    * mousestart, mousemoveなどのイベントハンドリング時に使用する<br>
 81    * $areaで示したブロック(jQueryオブジェクト)の左上を[0,0]とする
 82    * @param ev 取得に使用するイベントオブジェクト
 83    * @param ev 取得に使用するイベントオブジェクト
 84    * @return [x,y]で示す配列
 85    */
 86   Input.get_xy_mouse = function($area, ev){
 87     var offset = $area.offset();
 88     return [ev.pageX - offset.left, ev.pageY - offset.top];
 89   };
 90 
 91   /**
 92    * 現在タッチしている位置を取得する<br>
 93    * touchstart, touchmoveなどのイベントハンドリング時に使用する<br>
 94    * $areaで示したブロック(jQueryオブジェクト)の左上を[0,0]とする
 95    * @param ev 取得に使用するイベントオブジェクト
 96    * @param ex 取得に使用するイベントオブジェクト
 97    * @return [x,y]で示す配列
 98    */
 99   Input.get_xy_touch = function($area, ev){
100     var offset = $area.offset();
101     var touch = ev.originalEvent.targetTouches[0];
102     return [touch.pageX - offset.left, touch.pageY - offset.top];
103   };
104 
105   /**
106    * 現在全ての指がタッチしている位置を取得する<br>
107    * touchstart, touchmoveなどのイベントハンドリング時に使用する<br>
108    * $areaで示したブロック(jQueryオブジェクト)の左上を[0,0]とする
109    * @param ev 取得に使用するイベントオブジェクト
110    * @param ex 取得に使用するイベントオブジェクト
111    * @return [x,y]で示す配列の配列
112    */
113   Input.get_xy_multitouch = function($area, ev){
114     var result = [];
115     var offset = $area.offset();
116     var touches = ev.originalEvent.targetTouches;
117     for(var i=0; i<touches.length; i++){
118         var touch = touches[i];
119         result.push([touch.pageX - offset.left, touch.pageY - offset.top]);
120     }
121     return result;
122   };
123 
124   /**
125    * @property Windows 8以外のタッチデバイスの時はget_xy_touchを、それ以外はget_xy_mouseを使用する
126    */
127   Input.get_xy = ((Input.can_touch() && !(Input.is_windows8())) ? Input.get_xy_touch : Input.get_xy_mouse);
128 
129   window.m4w = $.extend({Input: Input}, window.m4w);
130 })(jQuery);
