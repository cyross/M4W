1 /**
  2  * @fileOverview Miyako4Web(Miyako for Web a.k.a M4W/m4w) Video Extension<br>
  3  * M4W用ビデオ対応拡張プラグイン
  4  * このファイルを読み込むと、window.m4wオブジェクトに以下のプロパティが追加される<br>
  5  * window.m4w.Video<br>
  6  * window.m4w.AssetsLoader.loaders.video<br>
  7  * AssetLoader.prototype.success関数の引数に以下のキーが追加される<br>
  8  * <table>
  9  * <tr><th>Key</th><th>Value</th></tr>
 10  * <tr><td>video</td><td>Videoオブジェクトの辞書オブジェクト</tr>
 11  * </table>
 12  * @example 実際のVideoを取得するときは以下の方法でアクセス可能<br>
 13  * 例1)<br>
 14  * function success(assets){<br>
 15  *    var video = assets.video.(id);<br>
 16  * }<br>
 17  * 例2)<br>
 18  * function success(assets){<br>
 19  *    var video = assets["video"]["id"];<br>
 20  * }<br>
 21  * AssetsLoader.load関数の引数も以下のように拡張される<br>
 22  * options.assets[].type には"video"と指定する
 23  * options.assets[].その他 にはVideoクラスコンストラクタの引数と同じものを指定
 24  *
 25  * @name Miyako for Web(M4W) Video extention
 26  * @author Cyross Makoto (サイロス誠)
 27  * @version 0.0.1
 28  * @revision 1
 29  * @require jQuery 1.9.0/2.0.0(or later)
 30  * @license MIT License (MITライセンス)
 31  */
 32 
 33 (function($){
 34   /**
 35    * @constructor
 36    * @class ビデオ(videoタグ)を管理・操作
 37    * @param id ビデオに固有に付けられるID(videoタグのidと共通にすればなお良い)
 38    * @param body 対応するvideoタグのjQueryオブジェクト
 39    */
 40   Video = function(id, body){
 41     this.id = id;
 42     this.body = body;
 43     this.is_play = false;
 44     this.is_pause = false;
 45   };
 46 
 47   /**
 48    * ビデオファイルを読み込む<br>videoタグを作り、再生ができるときに指定した関数を呼び出す
 49    * @example {id: "hoge", src: ["/hoge.mp4","/hoge.avi"]}
 50    * @param options.id videoタグに一意につけられるID
 51    * @param options.src ファイルのURLを配列で指定(マルチブラウザ対応、ブラウザ内では指定したどれかを再生)
 52    * @param [options.width] ビデオの横幅<br>ピクセル単位<br>省略時は640
 53    * @param [options.height] ビデオの高さ<br>ピクセル単位<br>省略時は480
 54    * @param [options.left] ブラウザ(もしくは親ブロックからの右方向の位置<br>単位はピクセル<br>省略時は0(ブロックの左端)
 55    * @param [options.top] ブラウザ(もしくは親ブロックからの上方向の位置<br>単位はピクセル<br>省略時は0(ブロックの上端)
 56    * @param [options.z] ビデオの奥行き<br>CSSのz-indexと同じ<br>省略時は0
 57    * @param [options.body] ビデオを埋め込むブロック<br>jQueryオブジェクトを指定<br>省略時はbodyブロック($("body"))
 58    * @param [options.loop] ビデオを繰り返し再生するかを指定<br>省略時はfalse(繰り返さない)
 59    * @param [options.autofinish] 再生終了時、自動的にvideoタグを取り除くかを指定<br>省略時はfalse(取り除かない)
 60    * @return ビデオを再生できる状態に持ち込んでいるDeferredオブジェクト<br>コールバック関数の引数は以下の内容のオブジェクト
 61    * <ul>
 62    *  <li>type: "video"</li>
 63    *  <li>id: options.id</li>
 64    *  <li>value: 生成したVideoオブジェクト</li>
 65    *  <li>options: 生成時のオプション</li>
 66    * </ul>
 67    */
 68   Video.load = function(options){
 69     var o = $.extend({
 70       id: this.id,
 71       width: 640,
 72       height: 480,
 73       left: 0,
 74       top: 0,
 75       z: 0,
 76       loop: false,
 77       autofinish: false,
 78       body: $("body")
 79     }, options);
 80 
 81     var $area = $("<video />");
 82     var defer = $.Deferred();
 83 
 84     $area.attr("id", o.id);
 85     $area.attr("preload", "metadata");
 86     $area.attr("loop", o.loop);
 87     $area.attr("width", o.width);
 88     $area.attr("height", o.height);
 89     $area.css("position", "absolute");
 90     $area.css("top", o.top);
 91     $area.css("left", o.left);
 92     $area.css("z-index", o.z);
 93 
 94     for(var i=0; i<o.src.length; i++){
 95       var $src = $("<source />");
 96       var url = o.src[i] + "?" + new Date().getTime();
 97       $src.attr("id", o.id+"_"+i);
 98       $src.attr("name", o.id);
 99       $src.attr("src", url);
100       $src.appendTo($area);
101     }
102     $area.appendTo(o.body);
103 
104     $area[0].addEventListener("loadedmetadata", (function(){
105       var d = defer;
106       var video_id = o.id;
107       var body = $area;
108       return function(){ d.resolve({type: "video", id: video_id, value: new Video(video_id, body), options: o}); };
109     }).bind(this)());
110 
111     if(o.autofinish == true){
112       $area[0].addEventListener("ended", (function(){
113         var d = defer;
114         var video_id = o.id;
115         var video = $area[0];
116         return function(){
117           video.finish();
118           d.resolve(video_id);
119         };
120       }).bind(this)());
121     }
122 
123     return defer.promise();
124   };
125 
126   /**
127    * ビデオを表示する
128    */
129   Video.prototype.show = function(){
130     $("video#"+this.id)[0].show();
131   };
132 
133   /**
134    * ビデオを非表示にする
135    */
136   Video.prototype.hide = function(){
137     $("video#"+this.id)[0].hide();
138   };
139 
140   /**
141    * ビデオの奥行きを設定する
142    */
143   Video.prototype.z = function(z){
144     var $v = $("video#"+this.id);
145     if(!z){ return $v.css("z-index"); }
146     $v.css("z-index", parseInt(z));
147   };
148 
149   /**
150    * ロードした音声の長さを返す<br>単位：秒
151    */
152   Video.prototype.length = function(){
153     return $("video#"+this.id)[0].duuration;
154   };
155 
156   /**
157    * 再生している位置を返す<br>単位：秒
158    */
159   Video.prototype.pos = function(){
160     return $("video#"+this.id)[0].currentTime;
161   };
162 
163   /**
164    * 音声のボリューム(0.0(無音)～1.0(最大))を返す
165    */
166   Video.prototype.volume = function(){
167     return $("video#"+this.id)[0].volume;
168   };
169 
170   /**
171    * 音声のボリューム(0.0(無音)～1.0(最大))を設定する
172    */
173   Video.prototype.set_volume = function(vol){
174     return $("video#"+this.id)[0].volume = vol;
175   };
176 
177   /**
178    * ビデオを最初から再生する
179    */
180   Video.prototype.play = function(){
181     var v = $("video#"+this.id)[0];
182     if(!this.is_pause && !v.ended){ v.pause(); }
183     v.play();
184     this.is_play = true;
185     this.is_pause = false;
186   };
187 
188   /**
189    * ビデオの再生を停止する<br>同時に、ビデオの再生位置を先頭に戻す
190    */
191   Video.prototype.stop = function(){
192     var v = $("video#"+this.id)[0];
193     if(v.ended){ return; }
194     if(!this.is_pause){ v.pause(); }
195     v.currentTime = 0;
196     this.is_play = false;
197     this.is_pause = false;
198   };
199 
200   /**
201    * ビデオの再生を停止する<br>同時に、ビデオの再生位置はそのまま
202    */
203   Video.prototype.pause = function(){
204     var v = $("video#"+this.id)[0];
205     if(this.is_pause || v.ended){ return; }
206     v.pause();
207     this.is_play = false;
208     this.is_pause = true;
209   };
210 
211 
212   /**
213    * 停止時の再生位置からビデオを再生する
214    */
215   Video.prototype.resume = function(){
216     $("video#"+this.id)[0].play();
217     this.is_play = true;
218     this.is_pause = false;
219   };
220 
221   /**
222    * videoタグを取り除く
223    */
224   Video.prototype.remove = function(){
225     $("video#"+this.id)[0].remove();
226   };
227 
228   /**
229    * 再生を停止して、videoタグを取り除く
230    */
231   Video.prototype.finish = function(){
232     this.pause();
233     this.hide();
234     this.remove();
235   };
236 
237   /**
238    * エラー状態かどうかを返す
239    * @return true/false
240    */
241   Video.prototype.is_error = function(){
242     return $("video#"+this.id)[0].error != null;
243   };
244 
245   window.m4w = $.extend({Video: Video}, window.m4w);
246 
247   /** @ignore */
248   window.m4w.AssetsLoader.loaders.video = function(options){
249     return Video.load(options);
250   };
251 })(jQuery);
