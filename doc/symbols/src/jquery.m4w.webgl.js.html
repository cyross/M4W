 17 (function($){
 18   /**
 19    * @constructor
 20    * @class スプライトを管理・操作
 21    * @param options.id スプライトに一意に一位につけられるID<br>省略時はvalue属性(Imageオブジェクト)のidプロパティ
 22    * @param options.value スプライトの元にするImageクラス(Javascript標準クラス)のオブジェクト
 23    * @param options.scene スプライトを追加するシーンオブジェクト
 24    * @param [options.x] ブロックの左端から右方向の位置(左端を0とする)<br>省略時は0
 25    * @param [options.y] ブロックの左端から右方向の位置(左端を0とする)<br>省略時は0
 26    * @param [options.gw] ジオメトリの幅<br>省略時はWebGLSprite.swと同じ値
 27    * @param [options.gh] ジオメトリの高さ<br>省略時はWebGLSprite.shと同じ値
 28    * @param [options.alphaTest] アルファテストのしきい値<br>省略時は0.5
 29    */
 30   WebGLSprite = function(options){
 31     var o = $.extend({
 32       id: options.value.id,
 33       x:0, y:0,
 34       gw:options.value.width, gh:options.value.height,
 35       alphaTest: 0.5 }, options);
 36     this.id = o.id;
 37     this.image = o.value;
 38     /** @property テクスチャ */
 39     this.texture = new THREE.Texture(this.image, new THREE.UVMapping());
 40     this.texture.needsUpdate = true;
 41     this.texture.sourceFile = this.image.src;
 42 
 43     /** @property マテリアル */
 44     this.material = new THREE.MeshBasicMaterial({map: this.texture, alphaTest: o.alphaTest});
 45 
 46     /** @property ジオメトリ */
 47     this.geometry = new THREE.PlaneGeometry(o.gw, o.gh, 4, 4);
 48 
 49     /** @property メッシュ */
 50     this.mesh = new THREE.Mesh(this.geometry, this.material);
 51 
 52     o.scene.add(this.mesh);
 53 
 54     /** @property 位置情報<br>ブロックの左端から右方向の位置(中心を0とする) */
 55     this.position = this.mesh.position;
 56 
 57     /** @property クオータニオン */
 58     this.quaternion = this.mesh.quaternion;
 59     this.mesh.useQuaternion = true;
 60     this.tmp_quaternion = new THREE.Quaternion();
 61 
 62     this.position.x = o.x;
 63     this.position.y = o.y;
 64   };
 65 
 66   /**
 67    * スプライトを移動させる<br>座標は、ブロックの中心が(0,0)で、値がプラスだと右・上方向、マイナスだと左・下方向となる
 68    * @param [options.x] 中心を0とした時の右方向の位置<br>省略時は現在位置
 69    * @param [options.y] 中心を0とした時の下方向の位置<br>省略時は現在位置
 70    * @param [options.z] 中心を0とした時の手前方向の位置<br>省略時は現在位置
 71    * @param [options.dx] 右方向をプラスとした時の移動量<br>省略時は0
 72    * @param [options.dy] 上方向をプラスとした時の移動量<br>省略時は0
 73    * @param [options.dz] 手前方向を0とした時の移動量<br>省略時は0
 74    */
 75   WebGLSprite.prototype.move = function(options){
 76     var o = $.extend({x: this.position.x, y: this.position.y, z: this.position.z, dx: 0, dy: 0, dz: 0}, options);
 77     this.position.x = o.x + o.dx;
 78     this.position.y = o.y + o.dy;
 79     this.position.z = o.z + o.dz;
 80   };
 81 
 82   /**
 83    * スプライトを回転させる<br>回転の中心は画像の中心<br>x,y,zの単位は度数(ラジアンではない)
 84    * @param x x軸方向の回転角度
 85    * @param y y軸方向の回転角度
 86    * @param z z軸方向の回転角度
 87    */
 88   WebGLSprite.prototype.rotate = function(x, y, z){
 89     this.tmp_quaternion.setFromEuler(new THREE.Vector3(x, y, z));
 90     this.quaternion.multiply(this.tmp_quaternion);
 91   };
 92 
 93   /**
 94    * ベクトルを渡してスプライトを回転させる<br>回転の中心は画像の中心<br>x,y,zの単位は度数(ラジアンではない)
 95    * @param vector THREE.Vector3クラスのベクトル。x,y,zはx軸・y軸・z軸それぞれの回転角度
 96    */
 97   WebGLSprite.prototype.rotatev = function(vector){
 98     this.tmp_quaternion.setFromEuler(vector);
 99     this.quaternion.multiply(this.tmp_quaternion);
100   };

102   var construct_inner = function(obj, $body, screen, screen_w, screen_h, options){
103     // シーン
104     obj.scene = new THREE.Scene();
105 
106     // レンダラ
107     obj.renderer = new THREE.WebGLRenderer({
108       canvas: screen.layer[0], // レイヤ(canvasタグ)を指定する
109       antialias: true, // アンチエイリアス有効
110       preserveDrawingBuffer: true
111     });
112     obj.renderer.setSize(screen_w, screen_h);
113 
114     obj.camera.position.z = options.z;
115 
116     obj.scene.add(obj.camera);
117 
118     $body.append(obj.renderer.domElement);
119   }
120 
121   WebGL = function($body, screen, screen_w, screen_h, options){
122     var _m4w = window.m4w;
123     var o = $.extend({
124       fov: 10,
125       near: 1,
126       far: 10000,
127       z: 10000
128     }, options);
129 
130     // カメラ
131     this.camera = new THREE.PerspectiveCamera(
132       o.fov,
133       screen_w / screen_h,
134       o.near,
135       o.far
136     );
137 
138     construct_inner(this, $body, screen, screen_w, screen_h, o);
139   };
140 
141   WebGL.screen_options = {get_context: false};
142 
143   WebGL.z_from_fov = function(fov, screen_h){
144     return screen_h / Math.tan(fov * Math.PI / 180.0) - 1;
145   }
146 
147   WebGL.prototype.render = function(context){
148     this.renderer.render(this.scene, this.camera);
149   }
150 
151   WebGLOrtho = function($body, screen, screen_w, screen_h, options){
152     var _m4w = window.m4w;
153     var o = $.extend({
154       near: 1,
155       far: 10000,
156       z: 1
157     }, options);
158 
159     // カメラ
160     this.camera = new THREE.OrthographicCamera(
161       screen_w / -2,
162       screen_w / 2,
163       screen_h / 2,
164       screen_h / -2,
165       o.near,
166       o.far
167     );
168 
169     construct_inner(this, $body, screen, screen_w, screen_h, o);
170   }
171 
172   WebGLOrtho.screen_options = WebGL.screen_options;
173 
174   WebGLOrtho.z_from_fov = WebGL.z_from_fov;
175 
176   WebGLOrtho.prototype.render = function(context){
177     this.renderer.render(this.scene, this.camera);
178   }

181   window.m4w = $.extend({WebGL: WebGL, WebGLOrtho: WebGLOrtho, WebGLSprite: WebGLSprite}, window.m4w);
182 })(jQuery);
