 17 (function($){
 18   // メートル/ピクセルの換算比率
 19   var SCALE = 1/30; // 1メートルあたり30ピクセル
 20 
 21   // ２次元配列の作成
 22   function createVector(x, y){
 23     return new Box2D.Common.Math.b2Vec2(x, y);
 24   }
 25 
 26   /**
 27    * @constructor
 28    * @class 物理演算を管理・操作するオブジェクトの生成
 29    * @param [options.grav_x] x方向の重力<br>省略時は0
 30    * @param [options.grav_y] y方向の重力<br>省略時は9.8
 31    * @param [options.doSleep] trueのときは、動かしていないオブジェクトを計算しない<br>省略時はtrue
 32    * @param [options.physics_time] stepメソッドを呼び出した時に進む時間(秒)<br>省略時は1/60(秒)
 33    * @param [options.velocity_iterations] stepメソッドを呼び出した時に行う速さ計算の回数<br>多くすれば精度は上がるが時間がかかる<br>省略時は5(回)
 34    * @param [options.position_iterations] stepメソッドを呼び出した時に行う位置計算の回数<br>多くすれば精度は上がるが時間がかかる<br>省略時は5(回)
 35    */
 36   Physics = function(options){
 37     var o = $.extend({
 38       grav_x: 0,
 39       grav_y: 9.8,
 40       doSleep: true,
 41       physics_time: 1/60,
 42       velocity_iterations: 5,
 43       position_iterations: 5
 44     }, options);
 45 
 46     // 物理シミュレーション関係
 47     this.physics_time = o.physics_time;
 48     this.velocity_iterations = o.velocity_iterations;
 49     this.position_iterations = o.position_iterations;
 50 
 51     // 物理世界の生成
 52     this.world = new Box2D.Dynamics.b2World(createVector(o.grav_x, o.grav_y), o.doSleep);
 53 
 54     this.body_list = [];
 55   }
 56 
 57   // ２次元配列の作成
 58   Physics.createVector = createVector;
 59 
 60   // メートル/ピクセルの換算比率
 61   Physics.SCALE = SCALE;
 62 
 63   Physics.bodyDefProperties = {"body_def_type": "type", "user_data": "userData", "allow_sleep": "allowSleep"};
 64   Physics.fixtureDefProperties = {"density": "density", "friction": "friction", "restitution": "restitution"};
 65 
 66   // 剛体定義の生成
 67   Physics.prototype.createBodyDef = function(options){
 68     var o = $.extend({
 69       body_def_type: Box2D.Dynamics.b2Body.b2_dynamicBody,
 70       allow_sleep: true,
 71       x: 0,
 72       y: 0,
 73       user_data: null}, options);
 74     var body_def = new Box2D.Dynamics.b2BodyDef();
 75 
 76     // 初期位置を設定
 77     body_def.position.Set(o.x*SCALE, o.y*SCALE);
 78     for(var p in Physics.bodyDefProperties){
 79       var name = Physics.bodyDefProperties[p];
 80       body_def[name] = o[p];
 81     }
 82 
 83     return body_def;
 84   }
 85 
 86   // フィクスチャ定義の生成
 87   Physics.prototype.createFixtureDef = function(shape, options){
 88     var o = $.extend({
 89       fixture_def: Box2D.Dynamics.b2FixtureDef,
 90       density: 0,
 91       friction: 0,
 92       restitution: 0
 93     }, options);
 94 
 95     var fixture_def = new o.fixture_def();
 96 
 97     fixture_def.shape = shape;
 98 
 99     for(var p in Physics.fixtureDefProperties){
100       var name = Physics.fixtureDefProperties[p];
101       fixture_def[name] = o[p];
102     }
103 
104     return fixture_def;
105   }
106 
107   // シェイプの生成
108   Physics.prototype.createShape = function(options){
109     var o = $.extend({ shape_def: Box2D.Collision.Shapes.b2PolygonShape, w:100, h:100 }, options);
110     var shape = new o.shape_def();
111 
112     // シェイプの大きさを設定
113     shape.SetAsBox((o.w/2)*SCALE, (o.h/2)*SCALE);
114 
115     return shape;
116   }
117 
118   // 剛体の生成
119   Physics.prototype.createBody = function(body_def, fixture_def){
120     var body = this.world.CreateBody(body_def);
121     body.CreateFixture(fixture_def);
122     return body;
123   }
124 
125   // 時間をすすめる
126   Physics.prototype.step = function(){
127     this.world.Step(this.physics_time, this.velocity_iterations, this.position_iterations);
128   }
129 
130   // 時間を進めた剛体の情報からuser_dataを更新する
131   Physics.prototype.update = function(){
132     for(var i=0, len=this.body_list.length; i<len; i++){
133       var body = this.body_list[i];
134       if(body == null){ continue; }
135       var data = body.getUserData();
136       if(data){ data.update(body); }
137     }
138   }
139 
140   function createInner(obj, physics, options){
141     var o = $.extend({}, options);
142     // シェイプを生成
143     obj.shape = physics.createShape(o);
144     // フィクスチャ定義を生成
145     obj.fixture_def = physics.createFixtureDef(obj.shape, o);
146     //  剛体を生成
147     obj.body = physics.createBody(obj.body_def, obj.fixture_def);
148 
149     physics.body_list.push(obj);
150   }
151 
152   /**
153    * @constructor
154    * @class 動く剛体を生成
155    * @param physics m4w.Physicsクラスのオブジェクト
156    * @param options.w 剛体の幅
157    * @param options.h 剛体の高さ
158    * @param [options.x] x方向の位置<br>省略時は0
159    * @param [options.y] y方向の位置<br>省略時は0
160    * @param [options.density] 剛体の密度(1立方メートルあたり質量densityキログラム)<br>省略時は0
161    * @param [options.friction] 剛体の摩擦係数<br>省略時は0
162    * @param [options.restitution] 剛体の弾性(反発係数)<br>省略時は0
163    * @param [options.user_data] 付帯させるデータ(要update()メソッド)<br>省略時はnull(update()メソッド不要)
164    * @param [options.body_def_type] 剛体の形式の数値<br>省略時はBox2D.Dynamics.b2Body.b2_dynamicBody
165    * @param [options.fixture_def] フィクスチャ定義のクラス<br>省略時はBox2D.Dynamics.b2FixtureDef
166    * @param [options.shape_def] シェイプのクラス<br>省略時はBox2D.Collision.Shapes.b2PolygonShape
167    * @param [options.allow_sleep] 動かないときは物理演算の対象から外すかを指定<br>省略時はtrue
168    */
169   DynamicBody = function(physics, options){
170     var o = $.extend({
171       body_def_type: Box2D.Dynamics.b2Body.b2_dynamicBody,
172       x: 0,
173       y: 0,
174       user_data: null}, options);
175     // 剛体定義を生成
176     this.body_def = physics.createBodyDef(o);
177     createInner(this, physics, o);
178   }
179 
180   DynamicBody.prototype.getUserData = function(){
181     return this.body.GetUserData();
182   }
183 
184   DynamicBody.prototype.getPosition = function(){
185     return this.body.GetPosition();
186   }
187 
188   // 移動方向を指定
189   DynamicBody.prototype.setLinearVelocity = function(vector){
190     this.body.SetLinearVelocity(vector);
191   }
192 
193   /**
194    * @constructor
195    * @class 固定剛体を生成
196    * @param physics m4w.Physicsクラスのオブジェクト
197    * @param options.w 剛体の幅
198    * @param options.h 剛体の高さ
199    * @param [options.x] x方向の位置<br>省略時は0
200    * @param [options.y] y方向の位置<br>省略時は0
201    * @param [options.density] 剛体の密度(1立方メートルあたり質量densityキログラム)<br>省略時は0
202    * @param [options.friction] 剛体の摩擦係数<br>省略時は0
203    * @param [options.restitution] 剛体の弾性(反発係数)<br>省略時は0
204    * @param [options.user_data] 付帯させるデータ(要update()メソッド)<br>省略時はnull(update()メソッド不要)
205    * @param [options.body_def_type] 剛体の形式の数値<br>省略時はBox2D.Dynamics.b2Body.b2_staticBody
206    * @param [options.fixture_def] フィクスチャ定義のクラス<br>省略時はBox2D.Dynamics.b2FixtureDef
207    * @param [options.shape_def] シェイプのクラス<br>省略時はBox2D.Collision.Shapes.b2PolygonShape
208    * @param [options.allow_sleep] 動かないときは物理演算の対象から外すかを指定<br>省略時はtrue
209    */
210   StaticBody = function(physics, options){
211     var o = $.extend({
212       body_def_type: Box2D.Dynamics.b2Body.b2_staticBody,
213       x: 0,
214       y: 0,
215       user_data: null}, options);
216     // 剛体定義を生成
217     this.body_def = physics.createBodyDef(o);
218     createInner(this, physics, o);
219   }
220 
221   StaticBody.prototype.getUserData = function(){
222     return this.body.GetUserData();
223   }
224 
225   StaticBody.prototype.getPosition = function(){
226     return this.body.GetPosition();
227   }
228 
229   window.m4w = $.extend({Physics: Physics, DynamicBody: DynamicBody, StaticBody: StaticBody}, window.m4w);
230 })(jQuery);
