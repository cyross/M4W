1 /**
  2  * @fileOverview Miyako4Web(Miyako for Web a.k.a M4W/m4w) Drawer Extension<br>
  3  * M4W用図形描画応拡張プラグイン
  4  * このファイルを読み込むと、window.m4wオブジェクトに以下のプロパティが追加される<br>
  5  * window.m4w.Drawer<br>
  6  *

 14 
 15 (function($){
 16   /**
 17    * @constructor
 18    * @class スプライトと同じタイミングで図形描画を管理・操作(ドローワー)
 19    * @param options.id スプライトに一意に一位につけられるID<br>省略時はvalue属性(Imageオブジェクト)のidプロパティ
 20    * @param [options.func] ブラウザ画面関数配列<br>関数の引数には対象ブロックのコンテキスト(context)が渡る<br>省略時は空の関数
 21    */
 22   Drawer = function(options){
 23     var o = $.extend({
 24       func: function(ctx){ }
 25     }, options);
 26     // 直接インスタンスオブジェクトにパラメータの内容を結合
 27     $.extend(this, o);
 28   };
 29 
 30   /**
 31    * 画面の図形の描画を行う
 32    */
 33   Drawer.prototype.render = function(ctx){
 34     this.func(ctx);
 35     return this;
 36   };
 37 
 38   window.m4w = $.extend({Drawer: Drawer}, window.m4w);
 39 })(jQuery);
