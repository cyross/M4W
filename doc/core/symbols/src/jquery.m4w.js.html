 30   /**
 31    * 画像などのアセットを読み込み、指定した関数をコールバックする<br>
 32    * すべてのロードに成功したときはsuccess関数、一つでも失敗したときはfailed関数を呼ぶ
 33    * success関数の引数は以下のキーを持つ辞書オブジェクト<br>
 34    * <table>
 35    * <tr><th>Key</th><th>Value</th></tr>
 36    * <tr><td>image</td><td>Imageオブジェクトの辞書オブジェクト</tr>
 37    * <tr><td>sprite</td><td>Spriteオブジェクトの辞書オブジェクト</tr>
 38    * </table>
 39    * failed関数の引数はdefer.when.failedメソッドの引数の配列が渡される<br>
 40    * @example 実際のImage/Sprite/Soundを取得するときは以下の方法でアクセス可能<br>
 41    * 例1)<br>
 42    * function success(assets){<br>
 43    *    var image = assets.image.(id);<br>
 44    *    var sprite = assets.sprite.(id);<br>
 45    * }<br>
 46    * 例2)<br>
 47    * function success(assets){<br>
 48    *    var image = assets["image"]["id"];<br>
 49    *    var sprite = assets["sprite"]["id"];<br>
 50    * }
 51    * @param options.assets アセット設定オブジェクトの配列
 52    * @param options.assets[].type アセットの形式<br>画像は"image"、スプライトは"sprite"と指定する
 53    * @param options.assets[].その他 Image,Spriteクラスコンストラクタの引数と同じものを指定
 54    * @param [options.success] ロードが全て完了したときに呼び出されるコールバック関数<br>引数はアセット管理オブジェクト<br>内容は後述
 55    * @param [options.failed] ロードに失敗した時に呼び出されるコールバック関数<br>引数は失敗した時に渡される引数リスト
 56    */
 57   AssetsLoader.load = function(options){
 58     var o = $.extend({
 59       success: function(assets){ return assets; },
 60       failed: function(args){ alert("error occured!"); return args; }
 61     }, options);
 62 
 63     var assets = {}
 64 
 65     for(var i=0; i<o.assets.length; i++){
 66       var asset = o.assets[i]
 67       assets[asset.id] = asset;
 68     }
 69 
 70     var exec_assets = [];
 71     for(var asset_id in assets){
 72       var asset = assets[asset_id];
 73       exec_assets.push((AssetsLoader.loaders[asset.type])(asset));
 74     }
 75     $.when.apply(this, exec_assets).then(function(){
 76       var loaded_assets = {
 77       };
 78       for(var i=0; i<arguments.length; i++){
 79         var asset = arguments[i];
 80         if (!(asset.type in loaded_assets)){ loaded_assets[asset.type] = {}; }
 81         loaded_assets[asset.type][asset.id] = asset.value;
 82       }
 83       o.success(loaded_assets);
 84     }).fail(function(){
 85       o.failed(arguments);
 86     });
 87   };

292   /**
293    * 画像ファイルを読み込む<br>imgタグを作り、再生ができるときに指定した関数を呼び出す
294    * @param options.id 画像に一意につけられるID
295    * @param options.src 対象ファイルのURLを配列で指定(マルチブラウザ対応)
296    * @return 画像を読み込んでいるDeferredオブジェクト<br>コールバック関数の引数は、{type: "image", id: options.id, value: 生成したImageオブジェクト}で示すオブジェクト
297    */
298   Sprite.load_image = function(options){
299     var defer = $.Deferred();
300     var img_id = options.id;
301     var img = new Image();
302     img.src = options.src+ "?" + new Date().getTime();
303     img.onload = function(){ defer.resolve({type: "image", id: img_id, value: img}); };
304     return defer.promise();
305   };
