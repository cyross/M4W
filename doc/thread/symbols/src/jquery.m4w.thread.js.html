1 /**
  2  * @fileOverview Miyako4Web(Miyako for Web a.k.a M4W/m4w) Thread Extension<br>
  3  * M4W用スレッド対応拡張プラグイン
  4  * このファイルを読み込むと、window.m4wオブジェクトに以下のプロパティが追加される<br>
  5  * window.m4w.ThreadManager<br>
  6  * window.m4w.Thread<br>
  7  *
  8  * @name Miyako for Web(M4W) Thread extention
  9  * @author Cyross Makoto (サイロス誠)
 10  * @version 0.0.1
 11  * @revision 1
 12  * @require jQuery 1.9.0/2.0.0(or later)
 13  * @license MIT License (MITライセンス)
 14  */

 16 (function($){
 17   /**
 18    * @constructor
 19    * @class 複数スレッドを一括管理・操作
 20    * @param [options.threads] Threadsクラスオブジェクトの配列<br>省略時は空配列
 21    * @param [options.interval] Threads.updateの呼び出し間隔<br>単位はミリ秒<br>省略時は10
 22    */
 23   ThreadManager = function(options){
 24     var o = $.extend({
 25       threads: [],
 26       interval: 10
 27     }, options);
 28    
 29     /**
 30      * IDとスレッドを関連付ける辞書オブジェクト
 31      */
 32     this.threads = o.threads;
 33     this.interval = o.interval;
 34     this.timer = null;
 35     /**
 36      * @property スレッド割り込みタイマー
 37      */
 38     this.thread_timer = null;
 39   };
 40 
 41   /**
 42    * @ignore
 43    */
 44   ThreadManager.prototype.update = function(){
 45     var ts = this.threads;
 46     for(var i=0; i<ts.length; i++){
 47       var th = ts[i];
 48       if(!('update' in th)){ continue; }
 49       if(th.is_exec && !th.waiting && !th.update()){ th.stop(); }
 50     }
 51   };
 52 
 53   /**
 54    * タイマーを使ったスプライト描画・スレッドの監視を開始する
 55    */
 56   ThreadManager.prototype.start = function(){
 57     this.threads.timer = setInterval(this.threads.update.bind(this.threads), this.threads.interval);
 58     var each_frame = requestAnimFrame(this.screen.interval);
 59     each_frame(this.screen.render.bind(this.screen));
 60     return this.body;
 61   };
 62 
 63   /**
 64    * スレッドの監視を停止する
 65    */
 66   ThreadManager.prototype.stop = function(){
 67     if(this.threads.timer != null){ clearInterval(this.threads.timer); }
 68     return this.body;
 69   };
 70 
 71   /**
 72    * @param [options.start] Thread#startで呼び出す関数(引数は同じ)
 73    * @param [options.update] Thread#updateで呼び出す関数(引数は同じ)<br>処理を継続するときはtrue、停止するときはfalseを返すこと
 74    * @param [options.stop] Thread#stopで呼び出す関数(引数は同じ)
 75    * @param [options.pause] Thread#pauseで呼び出す関数(引数は同じ)
 76    * @param [options.resume] Thread#resumeで呼び出す関数(引数は同じ)
 77    * @constructor
 78    * @class スレッドを実装
 79   */
 80   Thread = function(options){
 81     t = $.extend({
 82       start: function(params){ },
 83       update: function(){ return true; },
 84       stop: function(){ },
 85       pause: function(){ },
 86       resume: function(){ }
 87     }, options);
 88 
 89     this.is_exec = false;
 90     this.waiting = false;
 91     this.body = t;
 92   };
 93 
 94   /**
 95    * スレッドの開始処理
 96    * 開始時の処理内容をオーバーライドする
 97    * @param params スレッド開始時に渡したいオブジェクト
 98    */
 99   Thread.prototype.start = function(params){
100     this.body.start(params);
101     this.is_exec = true;
102   };
103 
104   /**
105    * スレッドのメイン処理
106    * 処理の内容をオーバーライドする
107    * 処理を継続するときはtrue、停止するときはfalseを返す
108    */
109   Thread.prototype.update = function(){
110     return this.body.update();
111   };
112 
113   /**
114    * スレッドの停止処理
115    * 停止時の処理内容をオーバーライドする
116    */
117   Thread.prototype.stop = function(){
118     this.is_exec = false;
119     this.body.stop();
120   };
121 
122   /**
123    * スレッドの中断
124    * ただし、停止時には中断しない
125    * 中断時の処理内容をオーバーライドする
126    */
127   Thread.prototype.pause = function(){
128     if(!this.is_exec){ return; }
129     this.waiting = true;
130     this.body.pause();
131   };
132 
133   /**
134    * スレッドの再開処理
135    * ただし、停止時には中断しない
136    * 再開時の処理内容をオーバーライドする
137    */
138   Thread.prototype.resume = function(){
139     if(!this.is_exec){ return; }
140     this.waiting = false;
141     this.body.resume();
142   };
143 
144   window.m4w = $.extend({ThreadManager: ThreadManager, Thread: Thread}, window.m4w);
145 })(jQuery);
